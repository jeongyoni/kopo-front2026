package java_studentManager5;

import java.util.ArrayList;
import java.util.Scanner;

public class StudentManager5 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        ArrayList<Student5> students = new ArrayList<>();

        boolean check = true;

        while (check) {
            System.out.println("--------------------------------------------------");
            System.out.println("1.학생 추가 | 2.학생 목록 | 3.학생 수정 | 4.학생 삭제 | 5.종료");
            System.out.println("--------------------------------------------------");
            System.out.print("선택> ");
            int select = Integer.parseInt(scanner.nextLine());

            switch (select) {
                case 1:
                    System.out.println();
                    System.out.println("학생 추가");
                    System.out.println("----------------");

                    System.out.print("학번(Key): ");
                    String studentID = scanner.nextLine();

                    boolean exists = false;
                    for (Student5 s : students) {
                        if (s.studentID.equals(studentID)) {
                            exists = true;
                            break;
                        }
                    }

                    if (exists) {
                        System.out.println("이미 존재하는 학번입니다.");
                        System.out.println();
                        break;
                    }

                    System.out.print("이름: ");
                    String name = scanner.nextLine();

                    System.out.print("나이: ");
                    int age = Integer.parseInt(scanner.nextLine());

                    System.out.print("전공: ");
                    String major = scanner.nextLine();

                    Student5 student = new Student5(studentID, name, age, major);
                    students.add(student);

                    System.out.println("학생을 추가하였습니다.");
                    System.out.println();
                    break;

                case 2:
                    System.out.println();
                    System.out.println("학생 목록");
                    System.out.println("----------------");

                    if (students.size() == 0) {
                        System.out.println("등록된 학생이 없습니다.");
                    } else {
                        for (Student5 s : students) {
                            System.out.println("학번: " + s.studentID +
                                    ", 이름: " + s.name +
                                    ", 나이: " + s.age +
                                    ", 전공: " + s.major);
                        }
                    }
                    System.out.println();
                    break;

                case 3:
                    System.out.println();
                    System.out.println("학생 수정");
                    System.out.println("----------------");

                    System.out.print("학번(Key): ");
                    String updateID = scanner.nextLine();

                    boolean foundUpdate = false;

                    for (Student5 s : students) {
                        if (s.studentID.equals(updateID)) {
                            System.out.print("새 이름: ");
                            s.name = scanner.nextLine();

                            System.out.print("새 나이: ");
                            s.age = Integer.parseInt(scanner.nextLine());

                            System.out.print("새 전공: ");
                            s.major = scanner.nextLine();

                            System.out.println("학생을 수정하였습니다.");
                            foundUpdate = true;
                            break;
                        }
                    }

                    if (!foundUpdate) {
                        System.out.println("해당 학번의 학생이 없습니다.");
                    }
                    System.out.println();
                    break;

                case 4:
                    System.out.println();
                    System.out.println("학생 삭제");
                    System.out.println("----------------");

                    System.out.print("학번(Key): ");
                    String deleteID = scanner.nextLine();

                    boolean foundDelete = false;

                    for (int i = 0; i < students.size(); i++) {
                        if (students.get(i).studentID.equals(deleteID)) {
                            students.remove(i);
                            System.out.println("학생을 삭제하였습니다.");
                            foundDelete = true;
                            break;
                        }
                    }

                    if (!foundDelete) {
                        System.out.println("해당 학번의 학생이 없습니다.");
                    }
                    System.out.println();
                    break;

                case 5:
                    System.out.println("프로그램 종료");
                    check = false;
                    break;

                default:
                    System.out.println("잘못된 번호를 입력했습니다.");
                    System.out.println();
                    break;
            }
        }

        scanner.close();
    }
}
